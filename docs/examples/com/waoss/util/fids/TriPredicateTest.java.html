19   package com.waoss.util.fids; 
20    
21   import org.junit.Test; 
22    
23   import static org.junit.Assert.assertEquals; 
24    
25   public class TriPredicateTest { 
26    
27       TriPredicate<Integer, Integer, Integer> triPredicate = (a, b, c) -> a == b && b == c; 
28    
29       @Test 
30       public void test() throws Exception {
31           assertEquals(true, triPredicate.test(5, 5, 5));
32       } 
33    
34       @Test 
35       public void and() throws Exception {
36           assertEquals(false, triPredicate.and((a, b, c) -> a == b && b > c).test(45, 41, 2));
37       } 
38    
39       @Test 
40       public void negate() throws Exception {
41           assertEquals(false, triPredicate.negate().test(5, 5, 5));
42       } 
43    
44       @Test 
45       public void or() throws Exception {
46           assertEquals(true, triPredicate.or((a, b, c) -> a == b && b == c).test(6, 6, 6));
47       } 
48    
49   }
